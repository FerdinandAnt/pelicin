# Display: Table

Use these `display` values to make an element hehave like a certain table element.

<FixedColumnWidthTable
  headers={[
    {
      key: 'display',
      text: <Code str="display: ..." />,
      formatContent: (str) => <Code str={str} />,
    },
    {
      key: 'correspondingTag',
      text: 'Makes it behaves like ...',
      formatContent: (str) => <Code str={`<${str}>`} />,
    },
  ]}
  contents={[
    { display: 'table', correspondingTag: 'table' },
    { display: 'table-row', correspondingTag: 'tr' },
    { display: 'table-cell', correspondingTag: 'td' },
    { '@separator': 'Row groups (table-xxx-group)' },
    { display: 'table-header-group', correspondingTag: 'thead' },
    { display: 'table-row-group', correspondingTag: 'tbody' },
    { display: 'table-footer-group', correspondingTag: 'tfoot' },
    { '@separator': 'Column groups (table-column-xxx)' },
    { display: 'table-column-group', correspondingTag: 'colgroup' },
    { display: 'table-column', correspondingTag: 'col' },
    { '@separator': 'Caption' },
    { display: 'table-caption', correspondingTag: 'caption' },
  ]}
/>
